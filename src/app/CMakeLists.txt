set(CMAKE_AUTOMOC ON)
set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(SOURCES
    trayicon/trayicon.cpp
    mainwindow/mainwindow.cpp
    main.cpp
)

set(HEADERS
    trayicon/trayicon.h
    mainwindow/mainwindow.h
)

qt5_add_resources(RESOURCES
    resources.qrc
)

qt5_wrap_ui(UI_HEADERS
    mainwindow/mainwindow.ui
)

set(INTERNAL_LIBS
    ${SOURCES}
    ${UI_HEADERS}
    ${HEADERS}
    ${MOC_HEADERS}
    ${RESOURCES}
)

set(EXTERNAL_LIBS
    Qt5::Widgets
    Qt5::DBus
    tuned-switcher-lib
)

add_executable(${GUI_NAME} ${INTERNAL_LIBS})
add_dependencies(${APPLET_NAME} ${I18N_NAME})
target_link_libraries(${GUI_NAME} PRIVATE ${EXTERNAL_LIBS})

target_compile_definitions(${GUI_NAME} PRIVATE
    TS_SCHEME_NAME=${SCHEME_NAME}
    TS_LAUNCHER_NAME=${GUI_LAUNCHER}
)

install(TARGETS ${GUI_NAME} RUNTIME DESTINATION "${CMAKE_INSTALL_BINDIR}" BUNDLE DESTINATION "${CMAKE_INSTALL_BINDIR}")
